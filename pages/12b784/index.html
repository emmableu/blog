def run_bandit(epsilon, steps=1000, runs=2000, k=10):
    rewards = np.zeros((runs, steps))
    optimal_action_counts = np.zeros((runs, steps))

    for run in range(runs):
        q_true = np.random.normal(0, 1, k)
        optimal_action = np.argmax(q_true)
        Q = np.zeros(k)
        N = np.zeros(k)

        for t in range(steps):
            action = epsilon_greedy(Q, epsilon)
            reward = np.random.normal(q_true[action], 1)

            Q, N = update_sample_average(Q, N, action, reward)

            rewards[run, t] = reward
            if action == optimal_action:
                optimal_action_counts[run, t] = 1

    return rewards.mean(axis=0), optimal_action_counts.mean(axis=0)
