no_win_key = "NOWIN"

def print_state(state):

def one_round(cur_arr, state):
    o_pos_idx = random.randint(0, len(cur_arr) - 1)
    o_pos = cur_arr[o_pos_idx]
    state[o_pos] = 'O'
    win = check_win(cur_arr, state)
    if win != no_win_key:
        return win
    del cur_arr[o_pos_idx]
    x_pos_idx = random.randint(0, len(cur_arr) - 1)
    x_pos = cur_arr[x_pos_idx]
    state[x_pos] = 'X'
    del cur_arr[x_pos_idx]
    win = check_win(cur_arr, state)
    if win != no_win_key:
        return win
    return no_win_key

def check_win(cur_arr, state, final_round=False):
    if final_round:
        state[cur_arr[0]] = 'O'
    win_pos_lst = [[0,1,2], [3,4,5], [6,7,8],
                   [0,3,6], [1,4,7], [2,5,8],
                   [0,4,8], [2,4,6]]
    for win_pos in win_pos_lst:
        if state[win_pos[0]] == state[win_pos[1]] == state[win_pos[2]]:
            if state[win_pos[0]] == 'X':
                return "X"
            if state[win_pos[0]] == 'O':
                return "O"
    return 'NOWIN'

def full_game():

    for _ in range(4):
        win = one_round(cur_arr, state)
        if win != no_win_key:
            return win

# ðŸ§  åŸºäºŽå¼ºåŒ–å­¦ä¹ ï¼ˆTD æ›´æ–°ï¼‰çš„ç­–ç•¥

import math
import random
from collections import defaultdict
import numpy as np
import torch
import torch.nn.functional as F
import matplotlib.pyplot as plt

no_win_key = "NOWIN"

def print_state(state):
    print("-----------")
    for idx in range(3):
        print("|", " ".join(state[idx * 3: (idx + 1) * 3]), "|")
    print("-----------")

def get_x_pos_idx(cur_arr, state, policy_dict):
    candidate_prob_lst = []
    for idx in range(len(cur_arr)):
        x_pos = cur_arr[idx]
        next_state = state.copy()
        next_state[x_pos] = "X"
        state_str = "".join(next_state)
        prob = policy_dict[state_str]
        candidate_prob_lst.append(prob)
    probs_tensor = torch.tensor(candidate_prob_lst)
    top_idx = torch.argmax(probs_tensor)
    return top_idx

def one_round(cur_arr, state, policy_dict, state_history):
    o_pos_idx = random.randint(0, len(cur_arr) - 1)
    o_pos = cur_arr[o_pos_idx]
    state[o_pos] = 'O'
    state_history.append(''.join(state))
    win = check_win(cur_arr, state)
    if win != no_win_key:
        return win, state_history
    del cur_arr[o_pos_idx]
    x_pos_idx = get_x_pos_idx(cur_arr, state, policy_dict)
    x_pos = cur_arr[x_pos_idx]
    state[x_pos] = 'X'
    del cur_arr[x_pos_idx]
    state_history.append(''.join(state))
    win = check_win(cur_arr, state)
    if win != no_win_key:
        return win, state_history
    return no_win_key, state_history

def check_win(cur_arr, state, final_round=False):
    if final_round:
        state[cur_arr[0]] = 'O'
    win_pos_lst = [[0,1,2], [3,4,5], [6,7,8],
                   [0,3,6], [1,4,7], [2,5,8],
                   [0,4,8], [2,4,6]]
    for win_pos in win_pos_lst:
        if state[win_pos[0]] == state[win_pos[1]] == state[win_pos[2]]:
            if state[win_pos[0]] == 'X':
                return "X"
            if state[win_pos[0]] == 'O':
                return "O"
    return 'NOWIN'

def full_game(policy_dict):
    cur_arr = list(range(9))
    state = [" " for _ in range(9)]
    state_history = []
    for _ in range(4):
        win, state_history = one_round(cur_arr, state, policy_dict, state_history)
        if win != no_win_key:
            return win, state_history
    return check_win(cur_arr, state, final_round=True), state_history

def update_policy_dict(win, state_history, policy_dict, log_dict):
    final_reward = {'X': 1, 'O': 0, 'NOWIN': 0.5}[win]
    prev_reward = final_reward
    td_err_lst = []
    for state in reversed(state_history):
        td_err = prev_reward - policy_dict[state]
        policy_dict[state] += 0.1 * td_err
        td_err_lst.append(td_err)
        prev_reward = policy_dict[state]
    log_dict['td_err'].append(max(td_err_lst))
    return policy_dict, log_dict

def calculate_win_rate(rounds, policy_dict=None):
    res = defaultdict(int)
    if policy_dict is None:
        policy_dict = defaultdict(lambda: 0.5)
    log_dict = defaultdict(list)
    for _ in range(rounds):
        win, state_history = full_game(policy_dict)
        policy_dict, log_dict = update_policy_dict(win, state_history, policy_dict, log_dict)
        res[win] += 1
    return res, log_dict

rounds = 1000
res, log_dict = calculate_win_rate(rounds)
print(res)
