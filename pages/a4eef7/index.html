132 - 1197-Minimum Knight Moves

LC link (opens new window)

# Solution

both below has time complexity O(|x|*|y|)

# 1. 双向bfs (preferred)

from collections import deque
class Solution:
    def minKnightMoves(self, target_x: int, target_y: int) -> int:
        q1 = set([(0,0)])
        q2 = set([(abs(target_x), abs(target_y))])
        if len(q1 & q2) > 0:
            return 0
        vis = q1 | q2
        step = 0
        while q1 and q2:
            if len(q1) > len(q2):
                q1, q2 = q2, q1
            temp = set()
            step += 1
            for x, y in q1:
                for dx, dy in [(-2,-1), (-2, 1), (-1, -2), (-1, 2), (1,-2), (1,2), (2,-1), (2,1)]:
                    i, j = x + dx, y + dy
                    if (i, j) in q2: # i,j 必须用括号包起来
                        # 必须在这里check，不然的话即将重合的点进不去
                        return step
                    if (i, j) not in vis:
                        temp.add((i, j))
                        vis.add((i, j))
            q1 = temp
        return -1

# 2. Simble BFS

class Solution:
    def minKnightMoves(self, target_x: int, target_y: int) -> int:
        q = collections.deque()
        q.append([0,0,0])
        visited = set()
        visited.add((0,0))
        while q:
            x,y,steps = q.popleft()
            if x==abs(target_x) and y==abs(target_y): #steps for all directions are the same.
                return steps
            dirs = [(x-1,y-2),(x-2,y-1),(x-2,y+1),(x-1,y+2),(x+1,y-2),(x+2,y-1),(x+1,y+2),(x+2,y+1)]
            for i,j in dirs:
                if (i,j) not in visited:
                    q.append([i,j,steps+1])
                    visited.add((i,j))
        return -1
